p5.js GIF Animation Methods

Complete reference with code examples for controlling animated GIFs

numFrames() Documentation
Returns the total number of frames in the animated GIF. Useful for setting up loops and frame ranges.
let gif;

function preload()
{
  gif = loadImage('images/default.gif');
}

function setup()
{
  // Get total number of frames
  let totalFrames = gif.numFrames();
  console.log('This GIF has ' + totalFrames + ' frames');
}
getCurrentFrame() Documentation
Returns the index of the currently displayed frame (0-based). Essential for tracking animation progress.
function draw()
{
  image(gif, 0, 0);
  
  // Check which frame is currently showing
  let currentFrame = gif.getCurrentFrame();
  text('Frame: ' + (currentFrame + 1), 10, 20);
}
setFrame() Documentation
Jump to a specific frame by index (0-based). Perfect for creating frame scrubbing or navigation controls.
function keyPressed()
{
  if (key === 'ArrowRight')
  {
    // Go to next frame
    let current = gif.getCurrentFrame();
    let next = (current + 1) % gif.numFrames();
    gif.setFrame(next);
  }
}
play() Documentation
Starts or resumes GIF animation playback. The animation will continue from the current frame.
function mousePressed()
{
  // Start playing the GIF animation
  gif.play();
  console.log('Animation started');
}
pause() Documentation
Pauses GIF animation playback. The current frame remains displayed until play() is called again.
function keyPressed()
{
  if (key === ' ') // Spacebar
  {
    // Pause the animation
    gif.pause();
    console.log('Animation paused');
  }
}
delay() Documentation
Sets the delay between frames in milliseconds. Lower values = faster playback, higher values = slower.
function setup()
{
  // Set animation speed to 200ms per frame
  gif.delay(200);
  
  // Make it play very fast (50ms per frame)
  // gif.delay(50);
}
reset() Documentation
Resets the GIF to the first frame and stops playback. Useful for restarting animations from the beginning.
function doubleClicked()
{
  // Reset to first frame and stop
  gif.reset();
  console.log('GIF reset to first frame');
}
